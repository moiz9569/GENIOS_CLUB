const {
  time,
  loadFixture,
} = require("@nomicfoundation/hardhat-network-helpers");
const { anyValue } = require("@nomicfoundation/hardhat-chai-matchers/withArgs");
const { expect } = require("chai");
const { ethers } = require("hardhat");

describe("GeniosClub G3X7 auto level update", function () {
  let geniosClub;
  let token;
  let pool;
  let accounts;
  let user1;
  let referrer;

  before(async function () {
    this.timeout(100000000);

    accounts = await ethers.getSigners();
    referrer = accounts[0];
    user1 = accounts[1];

    const Token = await ethers.getContractFactory("MyToken");
    token = await Token.deploy();
    await token.deployed();

    const Pool = await ethers.getContractFactory("Pool");
    pool = await Pool.deploy(token.address);
    await pool.deployed();

    const GeniosClub = await ethers.getContractFactory("GeniosClubG3X7");
    geniosClub = await GeniosClub.deploy(
      referrer.address,
      token.address,
      pool.address,
      referrer.address,
      referrer.address
    );
    await geniosClub.deployed();
  });

  it("register a new user", async function () {
    this.timeout(100000000);

    const level = 1;

    await token.transfer(user1.address, "5000000000000000000");
    await token
      .connect(user1)
      .approve(geniosClub.address, "5000000000000000000");

    await geniosClub
      .connect(user1)
      .RegistrationExt(referrer.address, referrer.address);
    const user = await geniosClub.Users(user1.address);

    console.log(String(user.Id), user1.address);

    expect(user.Id).to.equal(1);
    expect(user.Ref).to.equal(referrer.address);
    expect(
      await geniosClub.usersActiveG3X7Levels(user1.address, level)
    ).to.equal(true);
  });

  it("not activated level 2 in that point", async function () {
    this.timeout(100000000);

    expect((await geniosClub.Users(user1.address)).Amount).to.equal(0);
    expect(await geniosClub.usersActiveG3X7Levels(user1.address, 2)).to.equal(
      false
    );
  });

  it("register 20 users with a user1", async function () {
    this.timeout(100000000);

    const level = 1;
    const numUsers = 20;
    let Refaddr = user1.address;

    for (let i = 2; i < numUsers; i++) {
      const user = accounts[i];

      const levelPrice = "5000000000000000000";
      await token.mint(user.address, levelPrice);

      await token.connect(user).approve(geniosClub.address, levelPrice);
      await geniosClub.connect(user).RegistrationExt(Refaddr, Refaddr);

      const userData = await geniosClub.Users(user.address);
      expect(userData.Ref).to.equal(Refaddr);

      console.log(
        String(userData.Id),
        user.address,
        await geniosClub.usersActiveG3X7Levels(Refaddr, 2),
        String((await geniosClub.usersG3X7Matrix(Refaddr, 1)).ReinvestCount),
        String((await geniosClub.Users(Refaddr)).Amount)
      );

      Refaddr = user.address;

      expect(
        await geniosClub.usersActiveG3X7Levels(user.address, level)
      ).to.equal(true);
    }
  });

  it("activated level 2 in that point", async function () {
    console.log(await geniosClub.usersG3X7Matrix(user1.address, 1));
  });

  //   it("activated level 2 in that point", async function () {

  //     expect(await geniosClub.usersActiveG3X7Levels(user1.address, 2)).to.equal(
  //       true
  //     );
  //   });
});
